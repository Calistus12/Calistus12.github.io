Sample SQL Queries


-- Example query to analyze customer ride patterns
SELECT 
    user_type,
    DAYNAME(start_time) AS day_of_week,
    COUNT(*) AS number_of_rides,
    AVG(TIMESTAMPDIFF(MINUTE, start_time, end_time)) AS avg_ride_duration
FROM 
    cyclistic_rides
GROUP BY 
    user_type, DAYNAME(start_time)
ORDER BY 
    user_type, number_of_rides DESC;

-- Query to identify popular stations
SELECT 
    start_station_name,
    COUNT(*) AS number_of_starts,
    AVG(CASE WHEN user_type = 'member' THEN 1 ELSE 0 END) AS member_percentage
FROM 
    cyclistic_rides
GROUP BY 
    start_station_name
HAVING 
    number_of_starts > 1000
ORDER BY 
    number_of_starts DESC
LIMIT 10;
